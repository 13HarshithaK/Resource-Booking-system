/**
 * GENERATE PRINTABLE INVOICE VIEW
 * Creates a clean, printer-friendly invoice that can be saved as PDF

 * @param object $wpdb - WordPress database object
 * @return string - HTML for invoice page
 */
function generate_invoice_view($wpdb) {
    // GET PARAMETERS
    $customer_id = isset($_GET['customer_id']) ? intval($_GET['customer_id']) : 0;
    $view_type = isset($_GET['billing_view']) ? $_GET['billing_view'] : 'monthly';

    $year = isset($_GET['year']) ? intval($_GET['year']) : date('Y');
    $week_start = isset($_GET['week']) ? $_GET['week'] : null; // For weekly invoices
    
    // Validate customer ID
    if ($customer_id == 0) {
        return '<p>Invalid customer ID</p>';
    }
    
    // GET CUSTOMER INFORMATION
    $customer = $wpdb->get_row($wpdb->prepare("
        SELECT firstName, lastName, email
        FROM {$wpdb->prefix}amelia_users
        WHERE id = %d
    ", $customer_id));
    
    // CALCULATE DATE RANGE for invoice
    $first_day = sprintf('%04d-%02d-01', $year, $month);
    $last_day = date('Y-m-t', strtotime($first_day)); // Last day of month
    
    // If generating weekly invoice, adjust date range
    if ($view_type === 'weekly' && $week_start) {
        $first_day = $week_start;
        $last_day = date('Y-m-d', strtotime($week_start . ' +6 days'));
    }
    
    // GET BOOKINGS for this customer in date range
    $bookings = $wpdb->get_results($wpdb->prepare("
        SELECT 
            a.id,
            a.bookingStart,
            a.bookingEnd,
            a.serviceId,
            s.name as serviceName,
            s.price
        FROM {$wpdb->prefix}amelia_appointments a
        INNER JOIN {$wpdb->prefix}amelia_customer_bookings cb ON a.id = cb.appointmentId
        LEFT JOIN {$wpdb->prefix}amelia_services s ON a.serviceId = s.id
        WHERE cb.customerId = %d
        AND DATE(a.bookingStart) BETWEEN %s AND %s
        ORDER BY a.bookingStart ASC
    ", $customer_id, $first_day, $last_day));
    
    // CALCULATE TOTALS
    $total_cost = 0;
    $total_hours = 0;
    foreach ($bookings as $booking) {
        $start = strtotime($booking->bookingStart);
        $end = strtotime($booking->bookingEnd);
        $hours = ($end - $start) / 3600; // Convert seconds to hours

        $total_cost += floatval($booking->price); // Price from Amelia database
    }

                max-width: 800px;
                margin: 0 auto;
                padding: 20px;
                line-height: 1.6;
            }
            
            .invoice-header {
                border-bottom: 3px solid #0073aa;
                padding-bottom: 20px;
                margin-bottom: 30px;
            }
            
            .invoice-title {
                font-size: 32px;
                color: #0073aa;
                margin: 0;
            }
            
            .invoice-meta {
                display: flex;
                justify-content: space-between;
                margin-top: 20px;
            }
            
            .invoice-section {
                margin-bottom: 30px;
            }
            
            .section-title {
                font-size: 18px;
                color: #333;
                margin-bottom: 10px;
                border-bottom: 2px solid #eee;
                padding-bottom: 5px;
            }
            
            .invoice-table {
                width: 100%;
                border-collapse: collapse;
                margin-bottom: 20px;
            }
            
            .invoice-table th {
                background: #f5f5f5;
                padding: 12px;
                text-align: left;
                border-bottom: 2px solid #ddd;
            }
            
            .invoice-table td {
                padding: 10px 12px;
                border-bottom: 1px solid #eee;
            }
            
            .invoice-table tr:last-child td {
                border-bottom: none;
            }
            
            .total-section {
                background: #f9f9f9;
                padding: 20px;
                margin-top: 30px;
                border: 2px solid #0073aa;

/**
 * RENDER BILLING DATA (Summary and Bookings List)
 * Shows summary cards and detailed booking table
 * 
 * @param object $wpdb - WordPress database object
 * @param int $customer_id - Selected customer ID
 * @param string $view_type - 'monthly' or 'weekly'
 * @param int $month - Month number (1-12)
 * @param int $year - Year
 */
function render_billing_data($wpdb, $customer_id, $view_type, $month, $year) {
    // CALCULATE DATE RANGE
    $first_day = sprintf('%04d-%02d-01', $year, $month);
    $last_day = date('Y-m-t', strtotime($first_day));
    
    // GET BOOKINGS for this customer and date range
    $bookings = $wpdb->get_results($wpdb->prepare("
        SELECT 
            a.id,
            a.bookingStart,
            a.bookingEnd,
            a.serviceId,
            s.name as serviceName,
            s.price
        FROM {$wpdb->prefix}amelia_appointments a
        INNER JOIN {$wpdb->prefix}amelia_customer_bookings cb ON a.id = cb.appointmentId
        LEFT JOIN {$wpdb->prefix}amelia_services s ON a.serviceId = s.id
        WHERE cb.customerId = %d
        AND DATE(a.bookingStart) BETWEEN %s AND %s
        ORDER BY a.bookingStart ASC
    ", $customer_id, $first_day, $last_day));
    
    // If no bookings found, show message
    if (empty($bookings)) {
        return '<div class="no-data">No bookings found for this customer in ' . date('F Y', mktime(0, 0, 0, $month, 1, $year)) . '</div>';
    }
    
    // CALCULATE TOTALS
    $total_bookings = count($bookings);
    $total_hours = 0;
    $total_cost = 0;
    
    foreach ($bookings as $booking) {
        $start = strtotime($booking->bookingStart);
        $end = strtotime($booking->bookingEnd);
        $hours = ($end - $start) / 3600; // Duration in hours
        $total_hours += $hours;
        $total_cost += floatval($booking->price); // Sum up costs
    }
    
    ob_start();
    ?>
    
    <!-- SUMMARY CARDS -->
    <div class="billing-summary">
        <h3>Summary for <?php echo date('F Y', mktime(0, 0, 0, $month, 1, $year)); ?></h3>
        <div class="summary-grid">
            <!-- Total number of bookings -->
            <div class="summary-card">
                <div class="summary-label">Total Bookings</div>
                <div class="summary-value"><?php echo $total_bookings; ?></div>
            </div>
            <!-- Total hours booked -->
            <div class="summary-card">
                <div class="summary-label">Total Hours</div>
                <div class="summary-value"><?php echo number_format($total_hours, 1); ?></div>
            </div>
            <!-- Total cost -->
            <div class="summary-card">
                <div class="summary-label">Total Cost (CAD)</div>
                <div class="summary-value">$<?php echo number_format($total_cost, 2); ?></div>
            </div>
            <!-- Average cost per booking -->
            <div class="summary-card">
                <div class="summary-label">Average per Booking</div>
                <div class="summary-value">$<?php echo number_format($total_cost / $total_bookings, 2); ?></div>
            </div>
        </div>
    </div>
    
    <?php
    // DISPLAY APPROPRIATE VIEW (monthly list or weekly breakdown)
    if ($view_type === 'weekly') {
        echo render_weekly_billing($bookings, $month, $year);
    } else {
        echo render_monthly_billing($bookings);
    }
    
    return ob_get_clean();
}

/**
 * RENDER WEEKLY BILLING BREAKDOWN
 * Groups bookings by week and shows subtotal for each week
 * Each week has its own invoice generation button
 * 
 * @param array $bookings - Array of booking objects
 * @param int $month - Month number (for invoice links)
 * @param int $year - Year (for invoice links)
 */
function render_weekly_billing($bookings, $month, $year) {
    // GROUP BOOKINGS BY WEEK
    // Creates array: ['2024-02-12' => ['bookings' => [...], 'total' => 150.00], ...]
    $weeks = array();
    foreach ($bookings as $booking) {
        // Find Monday of the week this booking falls in
        $week_start = date('Y-m-d', strtotime('monday this week', strtotime($booking->bookingStart)));
        
        // Initialize week array if doesn't exist
        if (!isset($weeks[$week_start])) {
            $weeks[$week_start] = array(
                'bookings' => array(),
                'total' => 0
            );
        }
        
        // Add booking to this week
        $weeks[$week_start]['bookings'][] = $booking;
        $weeks[$week_start]['total'] += floatval($booking->price);
    }
    
    ob_start();
    
    // DISPLAY EACH WEEK
    foreach ($weeks as $week_start => $week_data):
        $week_end = date('Y-m-d', strtotime($week_start . ' +6 days'));
    ?>
    <div class="week-section">
        <!-- WEEK HEADER with date range and total -->
        <div class="week-header">
            <span>Week of <?php echo date('M d', strtotime($week_start)); ?> - <?php echo date('M d, Y', strtotime($week_end)); ?></span>
            <span class="week-total">
                $<?php echo number_format($week_data['total'], 2); ?> CAD
                <!-- Invoice button for this specific week -->
                <a href="?generate_invoice=1&customer_id=<?php echo $_GET['customer_id']; ?>&billing_view=weekly&month=<?php echo $month; ?>&year=<?php echo $year; ?>&week=<?php echo $week_start; ?>" 
                   target="_blank"
